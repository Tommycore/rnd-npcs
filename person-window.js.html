person-window.js

import { RndConf } from "./helper/configuration.js";

export class PersonWindow extends Application
{
  constructor()
  {
    super();
    this.data = createRandomPerson();
  }

  static get defaultOptions()
  {
    return mergeObject(super.defaultOptions,
    {
      popOut: true,
      template: `modules/${RndConf.SCOPE}/templates/person-generator.hbs`,
      title: game.i18n.localize('RNDNPCS.WINDOW-TITLE'),
      classes: ["form", "rnd-npcs"],
      width: 300
    });
  }

  /**
   * Creates a random Person
   * @returns 
   */
  static createRandomPerson()
  {
    return {name: "name"};
  }
}
